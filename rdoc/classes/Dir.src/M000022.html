  def Dir.rm_rf(dir)
    Dir.visit(dir, true) do |path|
      if FileTest.directory?(path)
  begin
    Dir.unlink(path)
  rescue # Security Exception for Content.IE
  end
      else
  begin
    File.unlink(path)
  rescue #Security exception index.dat etc.
  end
      end
    end
  end
