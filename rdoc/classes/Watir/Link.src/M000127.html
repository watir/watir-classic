        def initialize( ieController,  how , what )
            @ieController = ieController
            begin
                @o = ieController.getLink( how, what )
            rescue UnknownObjectException
                @o = nil
            end
            super( @o )
            property = {
             "innerText"        =>  "innerText" ,
             "href"             =>  "href" 
            }
            setProperties(property )
            deleteProperty("type")

            @how = how
            @what = what
        end
