        def link_string_creator
            n = []
            n <<   "href:".ljust(TO_S_SIZE) + self.href
            n <<   "inner text:".ljust(TO_S_SIZE) + self.innerText
            n <<   "img src:".ljust(TO_S_SIZE) + self.src if self.link_has_image
            return n
         end
