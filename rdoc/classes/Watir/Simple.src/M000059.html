        def assert_text_in_body(text,mesg=nil)
                if mesg.nil? then
                        assert(@@browser.pageContainsText(text), "couldn't find in body: [#{text}]")
                else
                        assert(@@browser.pageContainsText(text), mesg)
                end
        end
