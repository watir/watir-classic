        def assert_text_in_field(name, text, mesg=nil)
    if mesg.nil? then
      assert(@@browser.textField(:name, name).verify_contains(text), "couldn't find in field #{name}: [#{text}]")
    else
      assert(@@browser.textField(:name, name).verify_contains(text), mesg)
    end
        end
