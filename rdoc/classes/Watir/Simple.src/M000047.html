  def navigate_to_link_with_id(id)
    # FIXME: this should be moved into Watir!
    wait_before_and_after do
      doc = @@browser.getDocument()
      links = doc.links
      link = nil
      links.each do |n|
        match = false
        case id
        when Regexp
          match = (n.invoke("id") =~ id)
        when String
          match = (n.invoke("id") == id)
        end
        if match
          link = n
          break
        end
      end
      raise "Couldn't find link with id #{id}" unless link
      @@browser.goto link
    end
  end
