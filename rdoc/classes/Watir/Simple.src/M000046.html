  def navigate_to_link_with_url(url)
    # FIXME: this should be moved into Watir!
    wait_before_and_after do
      doc = @@browser.getDocument()
      links = doc.links
      link = nil
      links.each do |n|
        match = false
        case url
        when Regexp
          match = (n.invoke("href") =~ url)
        when String
          match = (n.invoke("href") == url)
        end
        if match
          link = n
          break
        end
      end
      raise "Couldn't find link with url #{url}" unless link
      @@browser.goto link
    end
  end
