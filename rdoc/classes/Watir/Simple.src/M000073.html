        def assert_image_with_id(id, mesg=nil)
                if mesg.nil? then
                        assert( get_image_with_id(id) != nil, "image with id: [#{id}] is not present")
                else
                        assert( get_image_with_id(id) != nil, mesg)
                end
        end
