        def initialize(ieController, how, what, parent_table=nil )
            @ieController = ieController
            @o= nil
            if how == :direct
                @o = what     # in this case, @o is the table body
            elsif how == :index
                @o=parent_table.bodies.get_IE_table_body_at_index( what )
            end
            @rows = []
            update_rows
            super(@o)
        end
