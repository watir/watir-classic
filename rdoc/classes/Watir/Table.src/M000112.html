        def [](index)
            raise UnknownTableException ,  "Unable to locate a table using #{@how} and #{@what} " if @o == nil

            elements = []
            for td in row(index).children
                if(td.children.length > 0) then 
                    elements << td.children(0)
                else
                    elements << td
                end
            end
            
            return TableRow.new(elements,@ieController)
        end
