        def show_links()

            props=["name" ,"id" , "href" , "innerText" ]
            doc = getDocument()
            s = ""
            doc.links.each do |n|
                props.each do |prop|
                    begin
                        p = n.invoke(prop)
                        s =s+ "  " + "#{prop}=#{p}".to_s.ljust(18)
                    rescue
                        # this object probably doesnt have this property
                    end
                end
                s=s+"\n"
                
            end
            puts  s

        end
