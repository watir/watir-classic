        def show_spans( )
            spans = getDocument().getElementsByTagName("SPAN")
            puts "Found #{spans.length} span tags"
            index=1
            spans.each do |d|
                puts "#{index}   id=#{d.invoke('id')}      style=#{d.invoke("className")}"
                index+=1
            end
        end
