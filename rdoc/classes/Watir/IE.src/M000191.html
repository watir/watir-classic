        def getObjectWithSrcOrAlt( what , how , container , htmlObjectTypes )
            o = nil
            container.each do |r|
                next unless o == nil
                begin
                    next unless htmlObjectTypes.include?(r.invoke("type").downcase)
                    case how
                    when :alt
                        attribute = r.alt
                    when :src
                        attribute = r.src
                    else
                        next
                    end
                    
                    o = r if what.matches( attribute )         
                    
                rescue
                end 
            end
            return o
        end
