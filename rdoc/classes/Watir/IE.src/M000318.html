        def show_tables( )
            tables = getDocument().getElementsByTagName("TABLE")
            puts "Found #{tables.length} tables"
            index=1
            tables.each do |d|
                puts "#{index}  id=#{d.invoke('id')}      rows=#{d.rows.length}   columns=#{d.rows["0"].cells.length }"
                index+=1
            end
        end
