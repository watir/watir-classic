        def getTablePart( part , how , what )
             doc = getDocument()
             parts = doc.all.tags( part )
             n = nil
             parts.each do | p |
                 next unless n==nil
                 if what.matches( p.invoke("id") )
                     n = p 
                 end
             end
             return n
        end
