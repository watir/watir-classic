    def getSelectedItems
        raise UnknownObjectException if @o==nil
        returnArray = []
        puts "There are #{@o.length} items"
        @o.each do |thisItem|
            if thisItem.selected == true
                puts "Item ( #{thisItem.text} ) is selected"
                returnArray << thisItem.text 
            end
        end
        return returnArray 
    end
