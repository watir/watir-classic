    def clickJSDialog_Thread(button = "OK" )

           puts "clickJSDialog_Thread Starting waiting.."
           sleep 3
           puts " clickJSDialog_Thread ... resuming"

               sleep 1
               clickWindowsButton("Microsoft Internet Explorer" , button )

           end

    end
