     def clickWindowsButton_hwnd (hwnd , buttonCaption )

         makeWindowActive(hwnd)

         d = getChildHandle( hwnd , buttonCaption )
         puts ("clickWindowsButton: handle for button: " + buttonCaption + " is " + d.to_s )

         if d != -1 
             makeWindowActive(hwnd)
             clickButtonWithHandle (d)
         else
             return false
         end

         return true
     end
