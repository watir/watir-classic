    def form( how , formName=nil )
        # If only one value is supplied, it is a form name
        if formName == nil
            formName = how
            formHow = :name
        else
            formName = formName
            formHow = how
        end
        log "form how is #{formHow} name is #{formName}"        
        return Form.new(self, formHow, formName)        
    end
