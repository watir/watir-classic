   def showAllObjects()
      log "-----------Objects in  page -------------" 
      doc = getDocument()
      s = ""
      props=["name" ,"id" , "value" , "alt" , "src"]
      doc.all.each do |n|
         begin
            s=s+n.invoke("type").to_s.ljust(16)
         rescue
            next
         end
         props.each do |prop|
            begin
               p = n.invoke(prop)
               s =s+ "  " + "#{prop}=#{p}".to_s.ljust(18)
            rescue
               # this object probably doesnt have this property
            end
         end
         s=s+"\n"
      end
      log s+"\n\n\n"
      clearFrame()
   end
