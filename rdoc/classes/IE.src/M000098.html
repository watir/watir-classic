   def button( how , what=nil  )
      if how.kind_of? Symbol
         raise MissingWayOfFindingObjectException   if what==nil
         b = Button.new(self, how , what )
      elsif how.kind_of? String
         log "how is a string - #{how}"
         b = Button.new(self, :caption, how)
      end
   end
