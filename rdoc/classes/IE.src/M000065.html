    def getImage( how, what )
        doc = getDocument()
        
        log "Finding an image how: #{how} What #{what}"
        
        images = doc.images
        o=nil
        images.each do |img|
            
            log "Image on page: src = #{img.src}"
            
            next unless o == nil
            
            case how
            when :src
                attribute = img.src
            when :name
                attribute = img.name
            when :id
                attribute = img.invoke("id")
            when :alt
                attribute = img.invoke("alt")
            else
                next
            end

            o = img if what.matches(attribute)
                        
        end # do
        return o
    end
