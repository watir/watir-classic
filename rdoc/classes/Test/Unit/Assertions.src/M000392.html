    def assert_arrayContains(array, string , message =nil)

         _wrap_assertion do
          assert_block("assert should not be called with a block.") { !block_given? }
          assert(array.kind_of?(Array) , "Must have an array")
          assert(array.include?(string) , message)
        end
    end
